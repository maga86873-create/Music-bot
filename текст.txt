python
import logging
from telegram import Update
from telegram.ext import ApplicationBuilder, MessageHandler, filters, ContextTypes.DEFAULT_TYPE, CommandHandler

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

BOT_TOKEN = "8312127782:AAGjB9TvhRaH1XkdvynqFCrlITj-0x4kjkU"
OWNER_ID = 7737234302

# Сохраняем последний chat_id пользователя
last_user_id = None

async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(
        "ТОпа, опять что-то сломал? Руки из жопы или просто день такой? Ладно, не ной, пиши, чо там у тебя отвалилось. Разберемся по-быстрому, если сам тупить не будешь. Жду инфу, блядь! 🛠️🔥"
    )

async def forward_to_owner(update: Update, context: ContextTypes.DEFAULT_TYPE):
    global last_user_id
    user = update.message.from_user
    last_user_id = update.message.chat.id
    
    # Отправляем тебе (владельцу)
    text = f"Бро, там кто-то плачется в личку. Видимо, руки всё-таки из жопы. Зайди, поставь бедолагу на путь истинный. 🛠️😏\n\nИмя: {user.full_name}\n@: @{user.username}\n\nСообщение: {update.message.text}"
    await context.bot.send_message(chat_id=OWNER_ID, text=text)
    
    # Отправляем пользователю подтверждение
    await update.message.reply_text(
        "*Всё, запульнул твою маляву! Сиди ровно и не дёргайся, скоро к тебе заглянем и всё обкашляем. 👍🔥*",
        parse_mode="Markdown"
    )

async def owner_reply(update: Update, context: ContextTypes.DEFAULT_TYPE):
    global last_user_id
    if update.message.from_user.id == OWNER_ID:
        if last_user_id:
            await context.bot.send_message(chat_id=last_user_id, text=update.message.text)
            await update.message.reply_text("✅ Ответ отправлен пользователю!")
        else:
            await update.message.reply_text("❌ Нет пользователя для ответа!")

app = ApplicationBuilder().token(BOT_TOKEN).build()
app.add_handler(CommandHandler("start", start))
app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND & ~filters.User(OWNER_ID), forward_to_owner))
app.add_handler(MessageHandler(filters.TEXT & filters.User(OWNER_ID), owner_reply))

print("🤖 Бот запущен!")
app.run_polling()
